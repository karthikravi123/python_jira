# GitHub Jira Integration using Python Flask

sudo apt update 

sudo apt install python3 python3-pip python3-venv

mkdir flask-app && cd flask-app 

python3 -m venv venv
source venv/bin/activate
pip3 install Flask 

python3 -m flask --version 


pip freeze > requirements.txt
